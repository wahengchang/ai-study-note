    ---
    import { getCollection, type CollectionEntry } from "astro:content";
    import BaseLayout from "../../layouts/BaseLayout.astro";
    import PostList from "../../components/PostList.astro";

    export async function getStaticPaths() {
      const posts = await getCollection("blog", ({ data }) => !data.draft);
      const categories = [...new Set(posts.map((p) => p.data.category))];
      return categories.map((category) => ({
        params: { category },
        props: {
          category,
          posts: posts
            .filter((p) => p.data.category === category)
            .sort((a, b) => b.data.pubDate.getTime() - a.data.pubDate.getTime()),
        },
      }));
    }

    interface Props {
      category: string;
      posts: CollectionEntry<"blog">[];
    }
    const { category, posts } = Astro.props;
    ---
    <BaseLayout title={`Category: ${category} — ai-study-note`} description={`Notes filed under ${category}.`}>
      <section>
        <h1 class="font-mono text-2xl text-text-primary mb-2">Category: <span class="text-brand-orange">{category}</span></h1>
        <p class="text-text-muted font-mono text-xs mb-6">{posts.length} posts</p>
        <PostList posts={posts} />
      </section>
    </BaseLayout>
    ```

    Rationale:
    - `getCollection` in `getStaticPaths` uses the MANDATORY draft filter.
    - `[...new Set(posts.map((p) => p.data.category))]` dedupes — each unique category becomes a route. Category values are already lowercased/trimmed at the schema level, so no further normalization.
    - Filter step inside `return categories.map(...)` passes the pre-filtered posts as a prop, avoiding a second `getCollection` call inside the component.
    - Sorted by `pubDate` desc for consistency with blog index.
    - No hardcoded `/blog/` or `/categories/` paths — PostList handles all blog post links via `${base}blog/...`.
  </action>

    ---
    import { getCollection, type CollectionEntry } from "astro:content";
    import BaseLayout from "../../layouts/BaseLayout.astro";
    import PostList from "../../components/PostList.astro";

    export async function getStaticPaths() {
      const posts = await getCollection("blog", ({ data }) => !data.draft);
      const tags = [...new Set(posts.flatMap((p) => p.data.tags))];
      return tags.map((tag) => ({
        params: { tag },
        props: {
          tag,
          posts: posts
            .filter((p) => p.data.tags.includes(tag))
            .sort((a, b) => b.data.pubDate.getTime() - a.data.pubDate.getTime()),
        },
      }));
    }

    interface Props {
      tag: string;
      posts: CollectionEntry<"blog">[];
    }
    const { tag, posts } = Astro.props;
    ---
    <BaseLayout title={`Tag: ${tag} — ai-study-note`} description={`Notes tagged #${tag}.`}>
      <section>
        <h1 class="font-mono text-2xl text-text-primary mb-2">Tag: <span class="text-brand-orange">#{tag}</span></h1>
        <p class="text-text-muted font-mono text-xs mb-6">{posts.length} posts</p>
        <PostList posts={posts} />
      </section>
    </BaseLayout>
    ```
